\begin{frame}{Fait 1 : le classeur des cascades, les briques conceptuelles}
\textbf{Une probabilité conceptuelle} (échelle : Très rare $\to$ Très probable).
\begin{itemize}
\item Propension du pilier d'amorce à démarrer une cascade (\texttt{ROOT}),
      puis force de chaque maillon \og $i$ entraîne $j$ \fg{} le long du chemin
      (\texttt{TRANS}, asymétrique).
\item Plus la chaîne est causalement plausible (amont $\to$ aval), plus elle est probable.
\end{itemize}
\medskip
\textbf{Une gravité conceptuelle} (échelle : Négligeable $\to$ Critique).
\begin{itemize}
\item Gravité propre de chaque pilier touché (\texttt{GBASE}), \emph{composée} le long
      de l'ordre : des défaillances qui s'enchaînent s'aggravent, des défaillances
      isolées ne se composent pas.
\end{itemize}
\medskip
\textbf{La criticité les croise} (échelle : Faible $\to$ Extrême) : c'est la matrice de
risque probabilité $\times$ gravité, appliquée à chaque cascade ordonnée.
\end{frame}

\begin{frame}{Du jugement aux catégories : les formules du barème}
\small
\textbf{Probabilité} $p$ /10 (produit des liens le long du chemin, borné à $[1,10]$) :
\[
p(a_1\!\to\!\cdots\!\to\!a_n) = \Big[\,10\cdot\texttt{ROOT}(a_1)\,
\textstyle\prod_{k} \texttt{TRANS}(a_k)(a_{k+1})\,\Big]
\]
\textbf{Gravité} $g$ /10 (étendue des dégâts, amplifiée par la cohérence du chemin) :
\[
\text{étendue} = \max_k \texttt{GBASE} + \kappa \!\sum_{\text{autres}}\! \texttt{GBASE},
\qquad
g = \Big[\,\text{étendue}\cdot\big(\alpha + \beta\,\overline{\texttt{TRANS}}_{\text{chemin}}\big)\Big]
\]
avec $\kappa = 0{,}40$, $\alpha = 0{,}50$, $\beta = 0{,}80$ (constantes de forme, non ancrées).
\medskip

\textbf{Criticité} : $c = \big[\sqrt{p \cdot g}\,\big]$ (le RPN de l'AMDEC ramené sur /10,
la racine ne change pas le classement), puis \textbf{paliers} $2/4/6/8$ :
$\le\!2$ Faible, $\le\!4$ Modérée, $\le\!6$ Élevée, $\le\!8$ Majeure, au-delà Extrême.
\medskip

\textbf{Les deux cadres orange de F1, recalculés :}
\begin{center}
\begin{tabular}{@{}l c c c c@{}}
\toprule
 & $p$ & $g$ & $\sqrt{p\cdot g}$ & criticité \\
\midrule
$1\!\to\!2$ & $[10 \times 1{,}00 \times 0{,}80] = 8$ & $[7{,}6 \times 1{,}14] = 9$ & $8{,}5$ & \textbf{Majeure} \\
$2\!\to\!1$ & $[10 \times 0{,}60 \times 0{,}20] = 1$ & $[7{,}6 \times 0{,}66] = 5$ & $2{,}2$ & \textbf{Faible} \\
\bottomrule
\end{tabular}
\end{center}
Le sens inverse est doublement puni : lien faible $\Rightarrow$ probabilité écrasée
\emph{et} amplification amortie (même étendue $7{,}6$).
\end{frame}

\begin{frame}{À quoi ça sert : probas conditionnelles au sein d'un même $N_{i,j}$}
$N_{i,j}$ : nombre d'incidents entrant par le pilier $j$ (segment $i$).
La cascade ne touche pas la fréquence, elle vit \textbf{dans la loi conditionnelle} :
\medskip
\begin{itemize}
\item sachant $N_{i,j} = n$, chacun des $n$ incidents part de $j$ et entraîne $j'$ avec
\[
\mathbb{P}(j' \mid j) \;=\; \frac{\texttt{TRANS}(j)(j')}{s_j},
\qquad s_j = \textstyle\sum_{k} \texttt{TRANS}(j)(k) ;
\]
\item l'incident touche donc un \textbf{ensemble} de piliers (pas un seul), et la perte
      de la cellule est une somme composée
      $S_{i,j} = \sum_{k=1}^{N_{i,j}} X_k$, chaque terme gonflé par la cascade.
\end{itemize}
\medskip
Conséquence : les lois de fréquence restent standard ; toute la dépendance entre piliers
est portée par le mécanisme. Estimable sur un registre horodaté :
$\widehat{\mathbb{P}}(j'|j) = N_{j,j'} / N_{j,\cdot}$.
\end{frame}

\begin{frame}{Fait 2 : refonte du Vasicek, le point de départ}
Le modèle structurel classique, \textbf{sans indice de pilier} : une latente de stress
par entité,
\[
X_i \;=\; \sqrt{\rho}\;Y \;+\; \sqrt{1-\rho}\;\varepsilon_i,
\qquad \mathrm{Var}(X_i) = 1,
\]
avec $Y$ le facteur systémique commun, $\varepsilon_i$ le choc propre, et l'incident
quand le stress franchit un seuil :
\[
D_i = \mathbf{1}\{X_i \ge K_i\}, \qquad K_i = \Phi^{-1}(1-\mathrm{PD}_i).
\]
\medskip
\textbf{Deux limites pour le cyber sous DORA :}
\begin{itemize}
\item la dépendance est \textbf{symétrique} (un seul $Y$, un seul $\rho$) : le modèle ne
      peut pas dire \og P1 entraîne P2 plus que l'inverse \fg, alors que le jugement
      d'expert est dirigé ;
\item le seuil $\Phi^{-1}(\mathrm{PD})$ suppose une PD stable et mesurable, qui n'existe
      pas en cyber.
\end{itemize}
\end{frame}

\begin{frame}{Généralisation A : la structure systémique dépend du pilier}
On indexe par entité $i$ \textbf{et pilier} $j$, avec un facteur systémique par pilier :
\[
X_{ij} \;=\; \sqrt{\rho_{ij}}\;Y_j \;+\; \sqrt{1-\rho_{ij}}\;\varepsilon_{ij}.
\]
\begin{itemize}
\item Chaque pilier a sa \og météo systémique \fg{} propre ($Y_j$), les $Y_j$ étant
      corrélés entre eux par une matrice $\Sigma_Y$ : les dépendances croisées vivent
      au niveau systémique.
\item Sur données rares, on ne calibre pas un $\rho_{ij}$ par entité : compromis
      hiérarchique $\rho_{ij} = \rho_j + u_{ij}$ (chaque entité gravite autour de la
      moyenne de son pilier).
\end{itemize}
\medskip
\textbf{Honnêteté} : cette partie A n'est pas neuve (multifacteur classique, employé dans
les stress-tests). La contribution propre commence en B.
\end{frame}

% =====================================================================
\begin{frame}{Généralisation B : la propagation dirigée}
On ajoute un terme de réseau \textbf{asymétrique} : le stress de $j$ subit celui des
autres piliers de la même entité,
\[
X_{ij} \;=\; \underbrace{\sum_{k\neq j} W_{jk}\,X_{ik}}_{\text{contagion dirigée}}
   \;+\; \sqrt{\rho_{ij}}\;Y_j \;+\; \sqrt{1-\rho_{ij}}\;\varepsilon_{ij},
\]
d'où la forme réduite, sur les cinq piliers d'une entité :
\[
\mathbf{X}_i \;=\; (I-W)^{-1}\,(\text{systémique} + \text{idiosyncratique}),
\qquad \text{sous réserve que } \rho(W) < 1.
\]
\begin{itemize}
\item $W$ est construite depuis \texttt{TRANS} : c'est la même information d'expert que
      le classeur des cascades, injectée dans le modèle quantitatif.
\item Emboîtement propre : $W = 0$ redonne la partie A ; un seul facteur redonne le
      Vasicek standard.
\end{itemize}
\end{frame}

% =====================================================================
\begin{frame}{La normalisation de Leontief : une stabilité garantie}
\textbf{Brancher \texttt{TRANS} telle quelle serait une erreur de catégorie} : son rayon
spectral dépasse un, la série $I + W + W^2 + \cdots$ diverge et l'inverse produit des
signes aberrants. \texttt{TRANS} n'a jamais été une matrice de parts.
\medskip
On applique la discipline du modèle entrées-sorties de Leontief : lire chaque ligne comme
des \textbf{parts} de stress reçu,
\[
\boxed{\;W \;=\; g \cdot \frac{\texttt{TRANS}}{\max_j \sum_k \texttt{TRANS}_{jk}}\;,
\qquad g \in (0,\,1]\;}
\]
où $g$ est la part de contagion du pilier le plus exposé. Alors, par Perron-Frobenius :
\[
\rho(W) \;\le\; \max_j \sum_k W_{jk} \;\le\; g \;<\; 1,
\]
et $(I-W)^{-1}$ existe. \textbf{La stabilité est garantie par construction}, sur tout le
domaine admissible, et non supposée. La normalisation par un scalaire préserve
l'hétérogénéité de ce que reçoit chaque pilier.
\end{frame}

% =====================================================================
\begin{frame}{Points bloquants}
\begin{enumerate}
\item \textbf{Calibrer $W$ (la matrice de propagation).} Aucune source publique ne le
      permet : les bases de pertes ne codent pas les piliers co-touchés, les bases de
      brèches n'ont pas l'horodatage fin. Il faudrait un \textbf{registre d'incidents
      horodaté} (format reporting DORA). En attendant : jugement d'expert $+$ analyses
      de sensibilité.
\item \textbf{Les trois jugements de base} (\texttt{ROOT}, \texttt{TRANS},
      \texttt{GBASE}) restent de l'expert. Les sensibilités montrent que le
      \emph{classement} des piliers y survit ; le \emph{niveau}, non. À assumer dans la
      rédaction.
\item \textbf{Le gain de contagion $g$} est un choix de modélisation non calibré :
      traité en axe de sensibilité, pas en paramètre estimé.
\end{enumerate}
\end{frame}
